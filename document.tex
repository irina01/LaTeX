\newtheorem{Th}{Теорема}
\begin{Th}[Виета]\label{thViet}
Если $x_1$~и~$x_2$~-— корни
квадратного уравнения, то
$x_1 + x_2 = -b$ и~$x_1 x_2 = c$.
\end{Th}
